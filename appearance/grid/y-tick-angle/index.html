    #include <matplot/matplot.h>

int main() {
    using namespace matplot;

    auto x = linspace(0, 50, 20);
    auto y = transform(x, [](double x) { return pow(x, 2); });
    stem(x, y);
    ytickangle(90);

    show();
    return 0;
}

    #include <matplot/matplot.h>

int main() {
    using namespace matplot;

    tiledlayout(2, 1);

    auto ax1 = nexttile();
    plot(ax1, rand(6, 6, 0, 1));

    auto ax2 = nexttile();
    plot(ax2, rand(6, 6, 0, 1));
    ytickangle(ax2, 45);

    show();
    return 0;
}
